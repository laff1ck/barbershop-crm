class Appointment(models.Model):
    STATUS_CHOICES = [
        ('planned',   'Запланирована'),
        ('completed', 'Завершена'),
        ('cancelled', 'Отменена'),
    ]

    client  = models.ForeignKey(
        'clients.Client', on_delete=models.PROTECT
    )
    master  = models.ForeignKey(
        'staff.Master', on_delete=models.PROTECT
    )
    service = models.ForeignKey(
        'services.Service', on_delete=models.PROTECT
    )
    scheduled_at = models.DateTimeField()
    status = models.CharField(
        max_length=20,
        choices=STATUS_CHOICES,
        default='planned'
    )
    notes = models.TextField(blank=True)

    class Meta:
        ordering = ['-scheduled_at']

    def __str__(self):
        return (f"{self.client} → {self.master}"
                f" {self.scheduled_at:%d.%m %H:%M}")

class Client(models.Model):
    # Уровни программы лояльности
    LOYALTY_LEVELS = [
        ('bronze', 'Бронза',  0,   0),
        ('silver', 'Серебро', 5,   5),
        ('gold',   'Золото',  10,  10),
        ('vip',    'VIP',     20,  15),
    ]

    name        = models.CharField(max_length=100)
    phone       = models.CharField(max_length=20, unique=True)
    email       = models.EmailField(blank=True)
    visit_count = models.PositiveIntegerField(default=0)
    loyalty_level = models.CharField(
        max_length=10, default='bronze'
    )
    discount_pct = models.PositiveSmallIntegerField(
        default=0
    )

    def recalculate_loyalty(self):
        # вызывается после каждой оплаты
        v = self.visit_count
        if   v >= 20: lvl, disc = 'vip',    15
        elif v >= 10: lvl, disc = 'gold',   10
        elif v >= 5:  lvl, disc = 'silver', 5
        else:          lvl, disc = 'bronze', 0
        self.loyalty_level = lvl
        self.discount_pct  = disc
        self.save(
            update_fields=['loyalty_level', 'discount_pct']
        )

Создание записи администратором

apps/appointments/views.py
apps/appointments/views.py AppointmentCreateView
Что делает: только администратор может создавать записи. После сохранения проверяется пересечение по времени мастера (конфликт слота). 
from django.contrib.auth.mixins import LoginRequiredMixin
from django.views.generic import CreateView
from django.urls import reverse_lazy
from .models import Appointment
from .forms  import AppointmentForm
from apps.accounts.mixins import AdminRequiredMixin

class AppointmentCreateView(AdminRequiredMixin, LoginRequiredMixin, CreateView):
    model         = Appointment
    form_class    = AppointmentForm
    template_name = 'appointments/appointment_form.html'
    success_url   = reverse_lazy('appointments:list')

    def form_valid(self, form):
        appointment = form.save(commit=False)
        # проверка конфликта слота мастера
        duration = appointment.service.duration_minutes
        end_time = appointment.scheduled_at + timedelta(minutes=duration)

        conflict = Appointment.objects.filter(
            master       = appointment.master,
            status       = 'planned',
            scheduled_at__lt = end_time,
            scheduled_at__gt = appointment.scheduled_at - timedelta(minutes=duration),
        ).exclude(pk=appointment.pk)

        if conflict.exists():
            form.add_error(None, 'Мастер занят в это время')
            return self.form_invalid(form)

        appointment.save()
        return super().form_valid(form)

class Payment(models.Model):
    METHOD_CHOICES = [
        ('cash', 'Наличные'),
        ('card', 'Карта'),
    ]

    appointment  = models.OneToOneField(
        'appointments.Appointment',
        on_delete=models.PROTECT
    )
    amount_gross = models.DecimalField(
        max_digits=8, decimal_places=2,
        verbose_name='Сумма до скидки'
    )
    discount_pct = models.PositiveSmallIntegerField(
        default=0,
        verbose_name='Скидка %'
    )
    amount_net = models.DecimalField(
        max_digits=8, decimal_places=2,
        verbose_name='Итоговая сумма'
    )
    method     = models.CharField(
        max_length=10, choices=METHOD_CHOICES
    )
    paid_at    = models.DateTimeField(auto_now_add=True)

    def save(self, *args, **kwargs):
        # пересчёт автоматически при сохранении
        disc = Decimal(self.discount_pct) / Decimal('100')
        self.amount_net = (self.amount_gross
                           * (Decimal('1') - disc)).quantize(
            Decimal('0.01')
        )
        super().save(*args, **kwargs)
apps/billing/views.py PaymentCreateView 
class PaymentCreateView(MasterOrAdminMixin, CreateView):
    model      = Payment
    form_class = PaymentForm

    def get_initial(self):
        appt = get_object_or_404(
            Appointment, pk=self.kwargs['pk']
        )
        client = appt.client
        return {
            'appointment':  appt,
            'amount_gross': appt.service.price,
            'discount_pct': client.discount_pct,
        }

    def form_valid(self, form):
        payment = form.save()
        appt    = payment.appointment
        client  = appt.client

        # 1. завершить запись
        appt.status = 'completed'
        appt.save(update_fields=['status'])

        # 2. прибавить визит и пересчитать лояльность
        client.visit_count += 1
        client.save(update_fields=['visit_count'])
        client.recalculate_loyalty()

        messages.success(
            self.request,
            f'Оплата {payment.amount_net} ₽ принята'
        )
        return redirect('appointments:list')

Автоматический пересчёт лояльности (сигнал)

apps/billing/signals.py
apps/billing/signals.py post_save → Payment
Как работает: Django-сигнал post_save срабатывает после каждого нового платежа. Автоматически обновляет счётчик визитов и уровень лояльности клиента без ручного вызова из view. 
from django.db.models.signals import post_save
from django.dispatch import receiver
from .models import Payment

@receiver(post_save, sender=Payment)
def update_client_loyalty(sender, instance, created, **kwargs):
    """После каждой новой оплаты пересчитываем лояльность клиента."""
    if not created:
        return   # обновление записи — не трогаем

    client = instance.appointment.client

    # атомарно увеличиваем счётчик визитов
    Client.objects.filter(pk=client.pk).update(
        visit_count=F('visit_count') + 1
    )
    client.refresh_from_db()

    # пороги: bronze 0, silver 5, gold 10, vip 20
    v = client.visit_count
    if   v >= 20: lvl, disc = 'vip',    15
    elif v >= 10: lvl, disc = 'gold',   10
    elif v >= 5:  lvl, disc = 'silver', 5
    else:          lvl, disc = 'bronze', 0

    Client.objects.filter(pk=client.pk).update(
        loyalty_level=lvl,
        discount_pct=disc
    )

Запрос расписания мастера (календарь)

apps/appointments/views.py
apps/appointments/views.py ScheduleView
Что делает: выдаёт все запланированные записи мастера на заданную неделю, сгруппированные по дням — для отображения в недельном календаре. 
from django.views.generic import TemplateView
from django.utils import timezone
from collections import defaultdict
from datetime import timedelta

class ScheduleView(LoginRequiredMixin, TemplateView):
    template_name = 'appointments/schedule.html'

    def get_context_data(self, **kwargs):
        ctx   = super().get_context_data(**kwargs)
        today = timezone.localdate()

        # начало недели (понедельник)
        week_start = today - timedelta(days=today.weekday())
        week_end   = week_start + timedelta(days=6)

        # если мастер — только свои записи
        qs = Appointment.objects.select_related(
            'client', 'service', 'master'
        ).filter(
            status='planned',
            scheduled_at__date__gte=week_start,
            scheduled_at__date__lte=week_end,
        )
        if self.request.user.role == 'master':
            qs = qs.filter(master__user=self.request.user)

        # группируем по дате
        by_day = defaultdict(list)
        for appt in qs.order_by('scheduled_at'):
            by_day[appt.scheduled_at.date()].append(appt)

        ctx['week_days']  = [week_start + timedelta(days=i) for i in range(7)]
        ctx['by_day']    = by_day
        ctx['week_start'] = week_start
        return ctx

apps/booking/views.py BookingView (публичная, без авторизации)
Особенность: клиент не авторизован. При записи автоматически ищется или создаётся объект Client по номеру телефона (get_or_create). 
from django.views import View
from django.shortcuts import render, redirect
from apps.clients.models      import Client
from apps.appointments.models import Appointment

class BookingView(View):
    """Доступна без авторизации — для клиентов барбершопа."""

    def get(self, request):
        return render(request, 'booking/booking_form.html', {
            'services': Service.objects.filter(is_active=True),
            'masters':  Master.objects.filter(is_active=True),
        })

    def post(self, request):
        data = request.POST

        # найти или создать клиента по телефону
        client, created = Client.objects.get_or_create(
            phone=data['phone'],
            defaults={'name': data['name']}
        )

        appt = Appointment.objects.create(
            client       = client,
            master       = get_object_or_404(Master, pk=data['master']),
            service      = get_object_or_404(Service, pk=data['service']),
            scheduled_at = parse_datetime(data['slot']),
            status       = 'planned',
        )

        return redirect('booking:confirm', pk=appt.pk)

Фильтрованная финансовая отчётность

apps/billing/views.py
apps/billing/views.py ReportView — агрегация Django ORM
Что делает: принимает GET-параметры date_from, date_to, master_id. Возвращает суммарную выручку, количество визитов и разбивку по мастерам через annotate + aggregate. 
from django.db.models import Sum, Count, Avg
from django.db.models.functions import TruncDate

class ReportView(AdminRequiredMixin, TemplateView):
    template_name = 'billing/report.html'

    def get_context_data(self, **kwargs):
        ctx = super().get_context_data(**kwargs)
        p   = self.request.GET

        qs = Payment.objects.select_related(
            'appointment__master',
            'appointment__client',
            'appointment__service',
        )

        # ── фильтрация ──────────────────────────────
        if p.get('date_from'):
            qs = qs.filter(paid_at__date__gte=p['date_from'])
        if p.get('date_to'):
            qs = qs.filter(paid_at__date__lte=p['date_to'])
        if p.get('master'):
            qs = qs.filter(appointment__master_id=p['master'])

        # ── агрегация ───────────────────────────────
        totals = qs.aggregate(
            total_revenue = Sum('amount_net'),
            total_visits  = Count('id'),
            avg_bill      = Avg('amount_net'),
        )

        # ── разбивка по мастерам ────────────────────
        by_master = (
            qs
            .values('appointment__master__user__get_full_name')
            .annotate(
                revenue = Sum('amount_net'),
                visits  = Count('id'),
            )
            .order_by('-revenue')
        )

        ctx.update({'totals': totals, 'by_master': by_master, 'payments': qs})
        return ctx
